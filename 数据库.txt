CREATE TABLE s_user (
  id bigint(225) NOT NULL auto_increment,
  name varchar(20)NOT NULL,
  login_name varchar(80)NOT NULL,
  login_passwd varchar(80)NOT NULL,
  PRIMARY KEY  (`id`)
);

CREATE TABLE s_role (
  id bigint(225) NOT NULL auto_increment,
  name varchar(20)NOT NULL,
  descption varchar(80)NOT NULL,
  PRIMARY KEY  (`id`)
);
